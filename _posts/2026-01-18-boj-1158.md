#include <iostream>
#include <queue>
#include <list>
using namespace std;

int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);
	
	int n, k;
	cin >> n >> k;
	list<int> answer;
	queue<int> people;
	
	// νμ— μ›μ† λ„£κΈ°
	for(int i = 0; i < n; ) {
		people.push(++i);
	}
	
	int count = 1;
	// νκ°€ λΉ λ• κΉμ§€
	while(!people.empty()) {
		if (count == k) {
			int plus = people.front();
			people.pop();
			answer.push_back(plus);
			count = 1;
		} else {
			count++;
			int pop = people.front();
			people.pop();
			people.push(pop);
		}
	}

    // μ¶λ ¥
    cout << "<";
    for (auto it = answer.begin(); it != answer.end(); it++) {
        if (it != answer.begin()) cout << ", ";
        cout << *it;
    }
    cout << ">\n";

    return 0;
}
