#include <iostream>
#include <vector>
using namespace std;

void dfs(int n, const vector<vector<int>>& adj, vector<bool>& visited) {
    visited[n] = true; 
	for(int j = 0; j < adj[n].size(); j++) {
    	int next = adj[n][j];
    	if(!visited[next]) {
    		visited[next] = true;
    		dfs(next, adj, visited);
	    }
	}
}
