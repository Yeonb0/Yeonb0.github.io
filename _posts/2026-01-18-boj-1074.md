#include <iostream>
using namespace std;

long long RC(int n, int r, int c) {
  if (n == 0) return 0;

  int half = 1 << (n - 1); // 2^(n-1)
  long long block = 1LL << (2 * n - 2);  // ν• μ‚¬λ¶„λ©΄μ ν¬κΈ°: half * half

  // 0λ² μ‚¬λ¶„λ©΄ (μΆμƒ)
  if (r < half && c < half)
    return RC(n - 1, r, c);
  // 1λ² μ‚¬λ¶„λ©΄ (μ°μƒ)
  if (r < half && c >= half)
    return block + RC(n - 1, r, c - half);
  // 2λ² μ‚¬λ¶„λ©΄ (μΆν•)
  if (r >= half && c < half)
    return 2 * block + RC(n - 1, r - half, c);
  // 3λ² μ‚¬λ¶„λ©΄ (μ°ν•)
  return 3 * block + RC(n - 1, r - half, c - half);
}

  ios::sync_with_stdio(false);
  cin.tie(0);

  int N, r, c;
  cin >> N >> r >> c;

  cout << RC(N, r, c);
  return 0;
}
